# Pin Circuit Event

Run with:
```bash
node eg/pin-circuit-event.js
```


```javascript
var five = require("johnny-five");

new five.Board().on("ready", function() {
  var pin = new five.Pin(5);

  // Event tests
  [ "high", "low" ].forEach(function( type ) {
    pin.on( type, function() {
      console.log( "Circuit Event: ", type );
    });
  });
});

```
