# Switch

<!--remove-end-->








Run this example from the command line with:
```bash
node eg/switch.js
```


```javascript
const {Board, Led, Switch} = require("johnny-five");
const board = new Board();

board.on("ready", () => {
  const spdt = new Switch(8);
  const led = new Led(13);
  
  spdt.on("open", () => led.off());
  spdt.on("close", () => led.on());
});

```
