# Color - EVShield EV3 (Raw)

<!--remove-end-->








Run this example from the command line with:
```bash
node eg/color-raw-EVS_EV3.js
```


```javascript
const {Board, Color} = require("johnny-five");
const board = new Board();

board.on("ready", () => {
  const color = new Color({
    controller: "EVS_EV3",
    mode: "RAW",
    pin: "BAS1"
  });

  color.on("change", () => {
    console.log("Color:");
    console.log("  rgb     : ", color.rgb);
    console.log("--------------------------------------");
  });
});

```
