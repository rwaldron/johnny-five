&nbsp;




Run this example from the command line with:
```bash
node eg/sensor-digital-microwave.js
```


```javascript
var five = require("johnny-five");
var board = new five.Board();

board.on("ready", () => {
  var microwave = new five.Sensor.Digital(7);

  microwave.on("change", () => {
    console.log(microwave.value);
  });
});

```
